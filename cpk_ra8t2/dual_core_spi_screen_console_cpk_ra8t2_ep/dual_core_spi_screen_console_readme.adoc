= RA8T2利用双核重定向输出到CPU1的LCD Console

RA8T2 包含两个 CPU 核心：CPU0(Cortex-M85 1GHz) 和 CPU1(Cortex-M33 250MHz)。在应用中可以将部分代码交由 CPU1 运行以让 CPU0 专注于关键计算任务。本示例在 spi_screen_console_cpknet_ra8t2_ep 的基础上进行修改，将屏幕控制和 C 标准输入输出重定向的任务由 CPU1 执行。

NOTE: 为不影响已有示例工程，本示例复制了一份 spi_screen_console_cpknet_ra8t2_ep 至 e2studio_llvm/cm85 下，并将工程重命名为 spi_screen_console_cm85

hal_entry() 函数内容

[source,c,subs="verbatim,attributes"]
----
void hal_entry(void)
{
	/* TODO: add your own code here */
#if CONSOLE_CFG_USE_UART == 0
	R_SCI_B_UART_Open(g_uart9.p_ctrl, g_uart9.p_cfg);
#endif

#if CONSOLE_CFG_USE_RTT
	CONSOLE_Init();
#endif
	LOG_Reset();

	s_lcd.orientation = SCREEN_ORIENTATION_VERTICAL;
	LCD_Init(&s_lcd);

	arm_2d_init();
	disp_adapter0_init();
#if CONSOLE_CFG_USE_LCD
	CONSOLE_Init();
	arm_2d_scene_player_set_switching_mode(&DISP0_ADAPTER, ARM_2D_SCENE_SWITCH_MODE_FADE_WHITE);
	arm_2d_scene_player_switch_to_next_scene(&DISP0_ADAPTER);
#endif

	R_GPT_Open(g_timer1.p_ctrl, g_timer1.p_cfg);
	R_GPT_Start(g_timer1.p_ctrl);

	R_IPC_Open(g_ipc0.p_ctrl, g_ipc0.p_cfg);
	R_IPC_Open(g_ipc1.p_ctrl, g_ipc1.p_cfg);
    /* 通知 CPU0，CPU1 就绪 */
	R_IPC_EventGenerate(g_ipc0.p_ctrl, IPC_GENERATE_EVENT_IRQ1);

	/* Wake up 2nd core if this is first core and we are inside a multicore project. */
#if (0 == _RA_CORE) && (1 == BSP_MULTICORE_PROJECT) && !BSP_TZ_NONSECURE_BUILD

#if BSP_TZ_SECURE_BUILD
    /* Take semaphore so 2nd core can clear it */
    R_BSP_IpcSemaphoreTake(&g_core_start_semaphore);
#endif

    R_BSP_SecondaryCoreStart();

#if BSP_TZ_SECURE_BUILD
    /* Wait for 2nd core to start and clear semaphore */
    while(FSP_ERR_IN_USE == R_BSP_IpcSemaphoreTake(&g_core_start_semaphore))
    {
        ;
    }
#endif
#endif

#if (1 == _RA_CORE) && (1 == BSP_MULTICORE_PROJECT) && BSP_TZ_SECURE_BUILD
    /* Signal to 1st core that 2nd core has started */
    R_BSP_IpcSemaphoreGive(&g_core_start_semaphore);
#endif

    while (1) {
    	CONSOLE_Task();
    	disp_adapter0_task();
    }

#if BSP_TZ_SECURE_BUILD
    /* Enter non-secure code */
    R_BSP_NonSecureEnter();
#endif
}

IPC1 和定时器的中断回调

[source,c,subs="verbatim,attributes"]
----
void IPC1_Callback(ipc_callback_args_t *p_args)
{
	switch (p_args->event) {
	case IPC_EVENT_IRQ0:
		CONSOLE_IPCIrqHandle();
		break;
	case IPC_EVENT_IRQ1:
		break;
	case IPC_EVENT_IRQ2:
		break;
	case IPC_EVENT_IRQ3:
		break;
	case IPC_EVENT_IRQ4:
		break;
	case IPC_EVENT_IRQ5:
		break;
	case IPC_EVENT_IRQ6:
		break;
	case IPC_EVENT_IRQ7:
		break;
	case IPC_EVENT_MESSAGE_RECEIVED:
		CONSOLE_IPCMsgHandle(p_args->message);
		break;
	case IPC_EVENT_FIFO_ERROR_EMPTY:
		LOG_W(__FUNCTION__, "IPC FIFO Empty");
		break;
	case IPC_EVENT_FIFO_ERROR_FULL:
		LOG_W(__FUNCTION__, "IPC FIFO Full");
		break;
	default:
		break;
	}
}

void TIMER1_Callback(timer_callback_args_t *p_args)
{
	if (p_args->event == TIMER_EVENT_CYCLE_END) {
		s_tick_ms++;
	}
}
